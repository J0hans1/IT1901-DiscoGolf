== Introduksjon

[plantuml]
....
component core {
	package todolist.core
	package todolist.json
}

component jackson {
}

todolist.json ..> jackson

component fxutil {
	package todolist.ui
}

component fxui {
	package todolist.ui
}


todolist.ui ..> todolist.core

	component fxml {
	}
}

}

}
